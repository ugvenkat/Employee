CREATE PROCEDURE [dbo].[usp_Add]
	@EmpID INT,
	@EmpName VARCHAR(20),
	@EmpPhoneNo VARCHAR(10)
AS
BEGIN
	SET NOCOUNT ON
	INSERT INTO Emp (EmpName, EmpPhoneNo) VALUES (@EmpName, @EmpPhoneNo)
END
--------------------------------------------------------------------------------------------------
GO
CREATE PROCEDURE [dbo].[usp_Edit]
	@EmpID INT,
	@EmpName VARCHAR(20),
	@EmpPhoneNo VARCHAR(10)
AS
BEGIN
	UPDATE Emp
	SET EmpName = @EmpName,
		EmpPhoneNo = @EmpPhoneNo
	WHERE EmpId = @EmpID
END
GO
---------------------------------------------------------------------------------------------
CREATE PROCEDURE [dbo].[usp_Delete]
	@EmpID INT
AS
BEGIN
	SET NOCOUNT ON;
	DELETE Emp
	WHERE EmpId = @EmpID
END
GO
---------------------------------------------------------------------------------------------
CREATE PROCEDURE [dbo].[usp_View]
	@EmpID INT
AS
BEGIN
	SET NOCOUNT ON;
	select * 
	FROM Emp
	WHERE EmpId = @EmpID
END
GO
---------------------------------------------------------------------------------------------
CREATE PROCEDURE [dbo].[usp_ViewAll]
AS
BEGIN
	SET NOCOUNT ON;
	SELECT * 
	FROM Emp
END
GO
